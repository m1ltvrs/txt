        async function sendAI() {
            const input = document.getElementById('aiInput');

            if (!msg) return;

        async function processComplexThought(input) {
            const rawOriginal = input.toLowerCase().trim();
            const raw = rawOriginal.replace(/[.,!?]/g, ''); 
            const mapNode = document.getElementById('currentMap');
            const addNode = (t) => { if(mapNode){const n=document.createElement('div'); n.className='neural-node'; n.innerText=t; mapNode.appendChild(n);}};
            
            if (/who are you|what are you|are you a bot|are you an ai|what can you do/.test(raw)) {
                addNode("IDENTITY");
                return { text: "I am Whirlpool AI! I can chat with you, solve complex math problems, tell you jokes, and look up facts or definitions using real-world sources. What do you want to explore?", source: "Whirlpool AI" };
            }
            if (/who made you|who created you|where did you come from/.test(raw)) {
                addNode("ORIGIN");
                return { text: "I was built for the Whirlpool Games platform to be your smart personal helper!", source: "Whirlpool AI" };
            }
            if (/^(hello|hi|hey|greetings|sup|howdy|yo|good morning|good afternoon|good evening|good night|heya)$/i.test(raw) || raw.includes("say hi")) {
                addNode("GREETING");
                return { text: "Hey there! How can I help you today? You can ask me a question, give me some math, or just chat.", source: "Whirlpool AI" };
            }
            if (/how are you|hows it going|how are things|how are you doing|whats up/.test(raw)) {
                addNode("CONVERSATION");
                return { text: "I'm running smoothly and ready to help! How are you doing?", source: "Whirlpool AI" };
            }
            if (/thank you|thanks|thx|appreciate it/.test(raw)) {
                addNode("POLITENESS");
                return { text: "You're very welcome! Let me know if you need anything else.", source: "Whirlpool AI" };
            }
            if (/i am (sad|bad|angry|mad|upset|depressed|tired|bored)/.test(raw) || /i feel (sad|bad|angry|mad|upset|depressed|tired|bored)/.test(raw)) {
                addNode("EMPATHY");
                return { text: "I'm sorry to hear that. Sometimes taking a quick break or playing a game helps! I can also tell you a joke if you want?", source: "Whirlpool AI" };
            }
            if (/i am (happy|good|great|excited|awesome|fine)/.test(raw) || /i feel (happy|good|great|excited|awesome|fine)/.test(raw)) {
                addNode("EMPATHY");
                return { text: "That's great to hear! I'm glad things are going well. What's on your mind today?", source: "Whirlpool AI" };
            }
            if (/^(ok|okay|cool|awesome|wow|nice|great|sweet|bet)$/i.test(raw)) {
                addNode("ACKNOWLEDGMENT");
                return { text: "Right on! What should we look at next?", source: "Whirlpool AI" };
            }
            if (/^(yes|yeah|yep|sure|why not)$/i.test(raw)) {
                addNode("AGREEMENT");
                return { text: "Awesome. Just tell me what you needâ€”whether it's a math problem, a definition, or a joke!", source: "Whirlpool AI" };
            }
            if (/^(no|nah|nope)$/i.test(raw)) {
                 addNode("DISAGREEMENT");
                 return { text: "No worries! Let me know when you're ready.", source: "Whirlpool AI" };
            }
            if (raw.includes("joke") || raw.includes("make me laugh") || raw.includes("funny")) {
                addNode("HUMOR DATABASE");
                try {
                    const res = await fetch('https://official-joke-api.appspot.com/random_joke');
                    const data = await res.json();
                    return { text: `${data.setup} ... **${data.punchline}**`, source: "Whirlpool Humor DB" };
                } catch(e) {
                    return { text: "Why do programmers prefer dark mode? Because light attracts bugs!", source: "Whirlpool Humor DB" };
                }
            }
            if (/[0-9+\-*/^=()\[\]]/.test(rawOriginal) || /sin|cos|tan|sqrt|log|pi|derivative|integral|matrix|sum/i.test(rawOriginal)) {
                addNode("MATH ENGINE");
                try {
                    const mathClean = rawOriginal.replace(/calculate|solve|what is|what's/gi, '').trim();
                    const res = await fetch(`https://api.mathjs.org/v4/?expr=${encodeURIComponent(mathClean)}`);
                    if (res.ok) return { text: `Computed result: **${await res.text()}**`, source: "MathJS Engine" };
                } catch (e) {}
            }
            const questionMatch = rawOriginal.match(/^(who|what|where|when|why)(?:\s+(?:is|are|was|were|did|do|does))?\s+(.*)/i);
            const defineMatch = rawOriginal.match(/^(define|meaning of)\s+(.*)/i);
            if (questionMatch || defineMatch) {
                let query = ""; let isWhy = false;
                if (questionMatch) { query = questionMatch[2]; if (questionMatch[1].toLowerCase() === 'why') isWhy = true; } 
                else if (defineMatch) { query = defineMatch[2]; }
                query = query.replace(/\?$/, '').trim(); 
                if (isWhy && query.length < 3) {
                    addNode("RECALLING CONTEXT");
                    const lastUserMsg = chatHistory.filter(m => m.type === 'user-msg').slice(-2, -1)[0]?.text || "that";
                    return { text: `Continuing our talk about "${lastUserMsg}": Could you specify what you'd like to know why about?`, source: "Conversational Memory" };
                }
                addNode("KNOWLEDGE BASE");
                try {
                    const wikiRes = await fetch(`https://en.wikipedia.org/api/rest_v1/page/summary/${encodeURIComponent(query)}`);
                    if (wikiRes.ok) {
                        const wikiData = await wikiRes.json();
                        if (wikiData.type !== "disambiguation" && wikiData.extract) {
                            const britLink = `\n\n*[Search Britannica for more on "${query}"](https://www.britannica.com/search?query=${encodeURIComponent(query)})*`;
                            return { text: wikiData.extract + britLink, source: "Wikipedia API & Britannica" };
                        }
                    }
                } catch(e) {}
                addNode("DICTIONARY API");
                try {
                    const dictQuery = query.split(' ')[0];
                    const dictRes = await fetch(`https://api.dictionaryapi.dev/api/v2/entries/en/${encodeURIComponent(dictQuery)}`);
                    if (dictRes.ok) {
                        const dictData = await dictRes.json();
                        if (dictData[0] && dictData[0].meanings[0]) {
                            const definition = dictData[0].meanings[0].definitions[0].definition;
                            return { text: `**${dictQuery}**: ${definition}`, source: "Free Dictionary API" };
                        }
                    }
                } catch(e) {}
            }
            if (raw === "why" || raw === "how" || raw.includes("tell me more")) {
                addNode("RECALLING CONTEXT");
                const lastUserMsg = chatHistory.filter(m => m.type === 'user-msg').slice(-2, -1)[0]?.text || "";
                return { text: `Continuing our talk about "${lastUserMsg}": What specifically would you like to know?`, source: "Conversational Memory" };
            }
            return { text: "I'm keeping track of this chat! I might not have the exact answer, but try asking me a 'who', 'what', 'where', 'when', or 'why' question, give me a math problem, or just say hi!", source: "Whirlpool AI" };
        }
